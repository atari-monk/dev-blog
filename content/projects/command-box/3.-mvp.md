## **Next.js + SQLite + Prisma + Meilisearch MVP (Simplified)**

### 1. Create Project (if not done)
```bash
npx create-next-app@latest command-box --typescript --use-pnpm
cd command-box
```

Choose defaults

- √ Would you like to use Tailwind CSS? ... No / Yes
- √ Would you like your code inside a `src/` directory? ... No / Yes
- √ Would you like to use App Router? (recommended) ... No / Yes
- √ Would you like to use Turbopack for `next dev`? ... No / Yes
- √ Would you like to customize the import alias (`@/*` by default)? ... No / Yes

### 2. Install Dependencies
```bash
pnpm add prisma @prisma/client meilisearch
pnpm add -D @types/node
```

### 3. Prisma Setup
```bash
pnpm dlx prisma init
```

Edit `prisma/schema.prisma`:
```prisma
datasource db {
  provider = "sqlite"
  url      = "file:./dev.db"
}

model Command {
  id        Int      @id @default(autoincrement())
  title     String
  command   String
  createdAt DateTime @default(now())
}
```

Create database:
```bash
pnpm prisma migrate dev --name init
```

Generate prisma:
```bash
pnpm prisma generate
```

### 6. API Route
`src/app/api/commands/route.ts`:
```ts
import prisma from "@/lib/prisma";
import meili from "@/lib/meilisearch";
import { NextResponse } from "next/server";

export async function GET(req: Request) {
  const { searchParams } = new URL(req.url);
  const q = searchParams.get("q");

  if (q) {
    const results = await meili.index("commands").search(q);
    return NextResponse.json(results.hits);
  }

  const commands = await prisma.command.findMany();
  return NextResponse.json(commands);
}

export async function POST(req: Request) {
  const { title, command } = await req.json();
  const newCmd = await prisma.command.create({ data: { title, command } });
  await meili.index("commands").addDocuments([newCmd]);
  return NextResponse.json(newCmd);
}

export async function PUT(req: Request) {
  const { id, title, command } = await req.json();
  const updatedCmd = await prisma.command.update({
    where: { id },
    data: { title, command }
  });
  await meili.index("commands").updateDocuments([updatedCmd]);
  return NextResponse.json(updatedCmd);
}

export async function DELETE(req: Request) {
  const { id } = await req.json();
  await prisma.command.delete({ where: { id } });
  await meili.index("commands").deleteDocument(id);
  return NextResponse.json({ success: true });
}
```

### 7. Frontend
Replace `src/app/page.tsx`:
```tsx
"use client";
import { useState, useEffect } from "react";

export default function Home() {
  const [commands, setCommands] = useState<any[]>([]);
  const [search, setSearch] = useState("");
  const [newCmd, setNewCmd] = useState({ title: "", command: "" });
  const [editingId, setEditingId] = useState<number | null>(null);
  const [editCmd, setEditCmd] = useState({ title: "", command: "" });

  const fetchCommands = async (q = "") => {
    const res = await fetch(`/api/commands${q ? `?q=${q}` : ""}`);
    setCommands(await res.json());
  };

  useEffect(() => { fetchCommands(); }, []);

  const addCommand = async (e: React.FormEvent) => {
    e.preventDefault();
    await fetch("/api/commands", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(newCmd),
    });
    setNewCmd({ title: "", command: "" });
    fetchCommands();
  };

  const startEditing = (cmd: any) => {
    setEditingId(cmd.id);
    setEditCmd({ title: cmd.title, command: cmd.command });
  };

  const updateCommand = async () => {
    if (!editingId) return;
    await fetch("/api/commands", {
      method: "PUT",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ id: editingId, ...editCmd }),
    });
    setEditingId(null);
    fetchCommands();
  };

  const deleteCommand = async (id: number) => {
    await fetch("/api/commands", {
      method: "DELETE",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ id }),
    });
    fetchCommands();
  };

  const copyToClipboard = (text: string) => {
    navigator.clipboard.writeText(text);
  };

  return (
    <div className="min-h-screen bg-gray-900 text-gray-100 p-6">
      <div className="max-w-2xl mx-auto">
        <h1 className="text-3xl font-bold mb-6 text-blue-400">Command Box</h1>

        <input
          value={search}
          onChange={(e) => { setSearch(e.target.value); fetchCommands(e.target.value); }}
          placeholder="Search commands..."
          className="bg-gray-800 border border-gray-700 p-2 w-full mb-6 rounded focus:outline-none focus:ring-2 focus:ring-blue-500"
        />

        <form onSubmit={addCommand} className="mb-8 p-4 bg-gray-800 rounded-lg">
          <h2 className="text-xl font-semibold mb-4 text-blue-400">Add New Command</h2>
          <input
            value={newCmd.title}
            onChange={(e) => setNewCmd({ ...newCmd, title: e.target.value })}
            placeholder="Title"
            className="bg-gray-700 border border-gray-600 p-2 w-full mb-3 rounded focus:outline-none focus:ring-2 focus:ring-blue-500"
            required
          />
          <textarea
            value={newCmd.command}
            onChange={(e) => setNewCmd({ ...newCmd, command: e.target.value })}
            placeholder="Command"
            className="bg-gray-700 border border-gray-600 p-2 w-full mb-3 rounded h-20 focus:outline-none focus:ring-2 focus:ring-blue-500"
            required
          />
          <button
            type="submit"
            className="bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded transition-colors"
          >
            Add Command
          </button>
        </form>

        <div className="space-y-4">
          {commands.map((cmd) => (
            <div key={cmd.id} className="border border-gray-700 p-4 rounded-lg bg-gray-800">
              {editingId === cmd.id ? (
                <div className="mb-4">
                  <input
                    value={editCmd.title}
                    onChange={(e) => setEditCmd({ ...editCmd, title: e.target.value })}
                    className="bg-gray-700 border border-gray-600 p-2 w-full mb-2 rounded focus:outline-none focus:ring-2 focus:ring-blue-500"
                  />
                  <textarea
                    value={editCmd.command}
                    onChange={(e) => setEditCmd({ ...editCmd, command: e.target.value })}
                    className="bg-gray-700 border border-gray-600 p-2 w-full mb-2 rounded h-20 focus:outline-none focus:ring-2 focus:ring-blue-500"
                  />
                  <div className="flex space-x-2">
                    <button
                      onClick={updateCommand}
                      className="bg-green-600 hover:bg-green-700 text-white px-3 py-1 rounded text-sm transition-colors"
                    >
                      Save
                    </button>
                    <button
                      onClick={() => setEditingId(null)}
                      className="bg-gray-600 hover:bg-gray-700 text-white px-3 py-1 rounded text-sm transition-colors"
                    >
                      Cancel
                    </button>
                  </div>
                </div>
              ) : (
                <>
                  <div className="flex justify-between items-start">
                    <h3 className="font-bold text-lg text-blue-300">{cmd.title}</h3>
                    <div className="flex space-x-2">
                      <button
                        onClick={() => startEditing(cmd)}
                        className="text-yellow-400 hover:text-yellow-300 text-sm"
                      >
                        Edit
                      </button>
                      <button
                        onClick={() => deleteCommand(cmd.id)}
                        className="text-red-400 hover:text-red-300 text-sm"
                      >
                        Delete
                      </button>
                    </div>
                  </div>
                  <pre 
                    onClick={() => copyToClipboard(cmd.command)}
                    className="bg-gray-700 p-3 rounded mt-2 overflow-x-auto cursor-pointer hover:bg-gray-600 transition-colors"
                  >
                    {cmd.command}
                  </pre>
                </>
              )}
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}
```
