## **6. API Route (App Router)**

Create **`src/app/api/commands/route.ts`**:

```ts
import { prisma } from "@/lib/prisma";
import { meili } from "@/lib/meilisearch";
import { NextResponse } from "next/server";

export async function GET(req: Request) {
  const { searchParams } = new URL(req.url);
  const q = searchParams.get("q");

  if (q) {
    const results = await meili.index("commands").search(q);
    return NextResponse.json(results.hits);
  }

  const commands = await prisma.command.findMany({
    orderBy: { createdAt: "desc" }
  });
  return NextResponse.json(commands);
}

export async function POST(req: Request) {
  const { title, command, tags } = await req.json();
  const newCmd = await prisma.command.create({
    data: { title, command, tags },
  });
  await meili.index("commands").addDocuments([newCmd]);
  return NextResponse.json(newCmd);
}

export async function DELETE(req: Request) {
  const { searchParams } = new URL(req.url);
  const id = searchParams.get("id");
  if (!id) return NextResponse.json({ error: "Missing id" }, { status: 400 });

  await prisma.command.delete({ where: { id: Number(id) } });
  await meili.index("commands").deleteDocument(Number(id));
  return NextResponse.json({ ok: true });
}
```

## **7. Frontend (App Router Page)**

Replace **`src/app/page.tsx`** with:

```tsx
"use client";
import { useEffect, useState } from "react";

type Command = {
  id: number;
  title: string;
  command: string;
  tags: string[];
};

export default function Home() {
  const [commands, setCommands] = useState<Command[]>([]);
  const [q, setQ] = useState("");
  const [form, setForm] = useState({ title: "", command: "", tags: "" });

  async function fetchCommands(search = "") {
    const res = await fetch(`/api/commands${search ? `?q=${search}` : ""}`);
    setCommands(await res.json());
  }

  useEffect(() => { fetchCommands(); }, []);

  async function addCommand(e: React.FormEvent) {
    e.preventDefault();
    await fetch("/api/commands", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        ...form,
        tags: form.tags.split(",").map(t => t.trim())
      }),
    });
    setForm({ title: "", command: "", tags: "" });
    fetchCommands();
  }

  async function deleteCommand(id: number) {
    await fetch(`/api/commands?id=${id}`, { method: "DELETE" });
    fetchCommands();
  }

  return (
    <div className="p-6 max-w-2xl mx-auto">
      <h1 className="text-2xl font-bold mb-4">Command Box</h1>

      <input
        value={q}
        onChange={e => { setQ(e.target.value); fetchCommands(e.target.value); }}
        placeholder="Search..."
        className="border p-2 w-full mb-4"
      />

      <form onSubmit={addCommand} className="mb-6 space-y-2">
        <input
          value={form.title}
          onChange={e => setForm({ ...form, title: e.target.value })}
          placeholder="Title"
          className="border p-2 w-full"
          required
        />
        <input
          value={form.command}
          onChange={e => setForm({ ...form, command: e.target.value })}
          placeholder="Command"
          className="border p-2 w-full"
          required
        />
        <input
          value={form.tags}
          onChange={e => setForm({ ...form, tags: e.target.value })}
          placeholder="Tags (comma separated)"
          className="border p-2 w-full"
        />
        <button className="bg-blue-600 text-white px-4 py-2 rounded">Add</button>
      </form>

      <ul className="space-y-2">
        {commands.map(cmd => (
          <li key={cmd.id} className="border p-3 rounded flex justify-between">
            <div>
              <strong>{cmd.title}</strong>
              <pre className="bg-gray-100 p-2 rounded mt-1">{cmd.command}</pre>
              <div className="text-sm text-gray-500">{cmd.tags.join(", ")}</div>
            </div>
            <div className="space-x-2">
              <button
                onClick={() => navigator.clipboard.writeText(cmd.command)}
                className="bg-green-500 text-white px-2 py-1 rounded"
              >
                Copy
              </button>
              <button
                onClick={() => deleteCommand(cmd.id)}
                className="bg-red-500 text-white px-2 py-1 rounded"
              >
                Delete
              </button>
            </div>
          </li>
        ))}
      </ul>
    </div>
  );
}
```
